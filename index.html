jAnimate
makes your animate.css rocks with jQuery
$('#jAnimate').animate('rubberBand infinite');
$('#jAnimate').animateEnd('rubberBand');
$('#jAnimate').animateAndEnd('flipOutY');
$('#jAnimate').animateAndCall('tada', function(self, effect){
	$(self).animateEnd(effect);
	alert('Animation was finished');
});
$('#jAnimate').animateAndCall('tada', function(self, effect){
	$(self).animateEnd(effect).animateAndEnd('pulse');
});
